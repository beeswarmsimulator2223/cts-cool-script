local function matchPart(name)
    if not name or type(name) ~= "string" then return nil end
    local lname = name:lower():gsub("%s+", "")
    if lname:find("seat") then return nil end

    if lname:find("driver") or lname:find("loader") or lname:find("gunner") or lname:find("commander") then
        return "Crew", ESP.CrewColor
    end

    if lname:find("ammu") or lname:find("amui") or lname:find("ammo") or lname:find("engine") or lname:find("fuel") or lname:find("transmission") or lname:find("radiator") then
        return "Module", ESP.ModuleColor
    end

    return nil
end

local function parseKeyCode(str)
    if not str or str == "" then return nil end
    local name = str:match("KeyCode%.(.+)") or str
    for _, item in ipairs(Enum.KeyCode:GetEnumItems()) do
        if item.Name == name then return item end
    end
    return nil
end

local function PenView_CreateUI()
    for _, v in ipairs(LocalPlayer.PlayerGui:GetChildren()) do
        if v.Name == "PenViewport" then v:Destroy() end
    end
    local sg = Instance.new("ScreenGui")
    sg.ResetOnSpawn = false; sg.IgnoreGuiInset = true; sg.DisplayOrder = -100
    sg.Name = "PenViewport"; sg.Parent = LocalPlayer.PlayerGui
    local vp = Instance.new("ViewportFrame", sg)
    vp.Size = UDim2.new(1,0,1,0); vp.BackgroundTransparency = 1
    vp.ImageTransparency = 0.25; vp.ZIndex = -100
    local cam = Instance.new("Camera"); vp.CurrentCamera = cam
    cam.CameraType = Enum.CameraType.Scriptable
    return {viewport = vp, vpcam = cam}
end

local function PenView_GetPenetration()
    local vehicles = Services.Workspace:FindFirstChild("Vehicles")
    if not vehicles then return 200 end
    local chassis = vehicles:FindFirstChild("Chassis" .. LocalPlayer.Name)
    if not chassis then return 200 end
    local gunFolder = chassis:FindFirstChild("Gun")
    if not gunFolder then return 200 end
    for _, gunWeapon in ipairs(gunFolder:GetChildren()) do
        local config = gunWeapon:FindFirstChild("Config")
        if config then
            local shells = config:FindFirstChild("Shells")
            if shells then
                for _, folder in ipairs(shells:GetChildren()) do
                    local penVal = folder:FindFirstChild("Penetration")
                    if penVal then return penVal.Value end
                end
            end
        end
    end
    return 200
end

local function PenView_FindGunBrick(chassis)
    local gun = chassis:FindFirstChild("Gun", true)
    if not gun then return nil end
    for _, obj in ipairs(gun:GetDescendants()) do
        if obj.Name == "GunBrick" then return obj end
    end
    return nil
end

local function PenView_GetArmorThickness(hitPart, hitPos, direction, hitNormal)
    if not hitPart or not hitPos then return 0 end
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Whitelist
    params.FilterDescendantsInstances = {hitPart}
    local result = Services.Workspace:Raycast(hitPos + direction * 4, -direction * 50, params)
    if result and result.Instance == hitPart then
        local thickness = (hitPos - result.Position).Magnitude / 0.00357
        if hitNormal then
            local cos = math.abs(hitNormal:Dot(-direction.Unit))
            thickness = thickness / math.max(cos, 0.05)
        end
        return thickness
    end
    return 0
end

local function PenView_UpdateViewport(ui, part, thickness, pen)
    if not ui or not ui.viewport or not ui.viewport.Parent then return end
    if not part or not part.Parent then
        if ui.viewport then ui.viewport:ClearAllChildren() end
        PenView.LastPart = nil; return
    end
    local mesh = ui.viewport:FindFirstChildWhichIsA("BasePart")
    if PenView.LastPart ~= part or not mesh then
        PenView.LastPart = part; ui.viewport:ClearAllChildren()
        local clone = part:Clone()
        clone.Transparency = 0.3; clone.CanCollide = false; clone.Anchored = true
        clone.Parent = ui.viewport; mesh = clone
    end
    if not mesh then return end
    local ok = pcall(function() mesh.CFrame = part.CFrame end)
    if not ok then ui.viewport:ClearAllChildren(); PenView.LastPart = nil; return end
    ui.vpcam.CFrame = Services.Workspace.CurrentCamera.CFrame
    ui.vpcam.FieldOfView = Services.Workspace.CurrentCamera.FieldOfView
    local color
    if thickness <= 0.1 or pen <= 0 then color = Color3.fromRGB(90,90,90)
    elseif thickness <= pen * 0.5 then color = Color3.fromRGB(0,255,0)
    elseif thickness < pen then
        local t = (thickness - pen*0.5) / (pen*0.5)
        color = Color3.new(1, 1-t, 0)
    else color = Color3.fromRGB(255,0,0) end
    mesh.Color = color
end

local function PenView_StartHeartbeat(ui)
    if PenView.HeartbeatConnection then PenView.HeartbeatConnection:Disconnect() end
    PenView.LastPart = nil
    PenView.HeartbeatConnection = Services.RunService.Heartbeat:Connect(function()
        if not ui or not ui.viewport or not ui.viewport.Parent then
            if PenView.HeartbeatConnection then PenView.HeartbeatConnection:Disconnect(); PenView.HeartbeatConnection = nil end; return
        end
        local vehicles = Services.Workspace:FindFirstChild("Vehicles")
        if not vehicles then if ui.viewport then ui.viewport:ClearAllChildren() end; return end
        local chassis = vehicles:FindFirstChild("Chassis" .. LocalPlayer.Name)
        if not chassis then if ui.viewport then ui.viewport:ClearAllChildren() end; PenView.LastPart = nil; PenView.LastChassisName = nil; return end
        if chassis.Name ~= PenView.LastChassisName then PenView.LastChassisName = chassis.Name; PenView.LastPart = nil end
        local gunBrick = PenView_FindGunBrick(chassis)
        if not gunBrick then return end
        local pen = PenView_GetPenetration()
        local origin = gunBrick.Position + gunBrick.CFrame.LookVector * 2
        local dir = gunBrick.CFrame.LookVector
        local rayParams = RaycastParams.new()
        rayParams.FilterType = Enum.RaycastFilterType.Blacklist
        rayParams.FilterDescendantsInstances = {chassis, Services.Workspace:FindFirstChild("Projectiles")}
        rayParams.IgnoreWater = true
        local result = Services.Workspace:Raycast(origin, dir * 3000, rayParams)
        if result and result.Instance and table.find(PenView.ArmorTypes, result.Instance.Name) and result.Instance.CanCollide then
            local thickness = PenView_GetArmorThickness(result.Instance, result.Position, dir, result.Normal)
            PenView_UpdateViewport(ui, result.Instance, thickness, pen)
        else
            if ui.viewport then ui.viewport:ClearAllChildren() end; PenView.LastPart = nil
        end
    end)
end

local function PenView_Reset()
    if PenView.HeartbeatConnection then PenView.HeartbeatConnection:Disconnect(); PenView.HeartbeatConnection = nil end
    PenView.LastPart = nil; PenView.LastChassisName = nil
    PenView.UI = PenView_CreateUI()
    PenView_StartHeartbeat(PenView.UI)
end

local function PenView_Monitor()
    task.spawn(function()
        while Other.PenView do
            local vehicles = Services.Workspace:FindFirstChild("Vehicles")
            local chassis = vehicles and vehicles:FindFirstChild("Chassis" .. LocalPlayer.Name)
            if chassis and not PenView.HeartbeatConnection then PenView_Reset()
            elseif not chassis and PenView.HeartbeatConnection then
                if PenView.UI and PenView.UI.viewport then PenView.UI.viewport:ClearAllChildren() end
            end
            task.wait(0.5)
        end
    end)
end

local function UpdateMarkPositions()
    if not Camera then return end; if not Mark.Enabled then return end
    local camPos = Camera.CFrame.Position; local vpSize = Camera.ViewportSize
    for _, espData in pairs(ESP.Instances) do
        if espData.MarkBillboard and espData.IsHull then
            local target = espData.TurretAdornee or espData.AdorneePart or espData.Target
            if target and target.Parent then
                local ok, pos = pcall(GetModelPosition, target)
                if not ok or not pos then
                    local fallback = espData.AdorneePart or espData.Target
                    pos = fallback:IsA("BasePart") and fallback.Position or (fallback:IsA("Model") and fallback:GetBoundingBox().Position) or fallback.Position
                end
                if (pos - camPos).Magnitude >= Mark.Distance then
                    local sp, onScreen = Camera:WorldToViewportPoint(pos)
                    if onScreen and sp.Z > 0 then
                        local x = math.clamp(sp.X - Mark.Size/2, 0, vpSize.X - Mark.Size)
                        local y = math.clamp(sp.Y - Mark.OffsetY - Mark.Size/2, 0, vpSize.Y - Mark.Size)
                        espData.MarkBillboard.Position = UDim2.new(0, x, 0, y)
                        espData.MarkBillboard.Visible = true
                    else espData.MarkBillboard.Visible = false end
                else espData.MarkBillboard.Visible = false end
            else espData.MarkBillboard.Visible = false end
        end
    end
end

local function ProcessInternalParts(chassis)
    -- Scan ALL descendants of this chassis; match using matchPart function
    for _, part in ipairs(chassis:GetDescendants()) do
        if part:IsA("BasePart") then
            if part.Transparency < 1 then
                local partType, color = matchPart(part.Name)
                if partType then
                    local isCrew = (partType == "Crew")
                    if not ESP.InternalInstances[part] then
                        local highlight = Instance.new("Highlight")
                        highlight.FillColor = color
                        highlight.FillTransparency = 0.15
                        highlight.OutlineColor = color
                        highlight.OutlineTransparency = 0.05
                        highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                        highlight.Adornee = part
                        highlight.Parent = ESPFolder

                        local enabled = isCrew and ESP.CrewEnabled or ESP.ModulesEnabled
                        highlight.Enabled = enabled and (part.Transparency < 1)

                        ESP.InternalInstances[part] = {
                            Highlight = highlight,
                            Color = color, IsCrew = isCrew
                        }
                    end
                end
            end
        end
    end
end

local function findHighlightTarget(model)
    if not model then return nil end
    for _, child in ipairs(model:GetDescendants()) do
        if child:IsA("Model") or child:IsA("Folder") then

function ProcessChassis(chassis)
    if not chassis:IsA("Model") or not chassis.Name:match("^Chassis") then return end
    local playerName = chassis.Name:match("^Chassis(.+)$")
    if not playerName or playerName == "" then return end
    if playerName == LocalPlayer.Name then return end

    local targetPlayer = Services.Players:FindFirstChild(playerName)
    local isTeammate = false
    if targetPlayer then
        local localTeam = LocalPlayer.Team; local targetTeam = targetPlayer.Team
        if localTeam and targetTeam and localTeam == targetTeam then isTeammate = true end
    end

    if ESP.TeamCheck and isTeammate then
        -- Remove existing highlight ESP for this chassis if it became a teammate
        local function tryRemove(obj)
            if not obj then return end
            local ed = ESP.Instances[obj]
            if not ed then
                local armor = findHighlightTarget(obj)
                if armor then ed = ESP.Instances[armor] end
            end
            if ed then
                if ed.Instance then ed.Instance:Destroy() end
                if ed.DistanceBillboard then ed.DistanceBillboard:Destroy() end
                if ed.MarkBillboard then ed.MarkBillboard:Destroy() end
                ESP.Instances[ed.Target] = nil
            end
        end
        -- Remove internal (crew/module) ESP for this chassis
        for part, iData in pairs(ESP.InternalInstances) do
            if part and part:IsDescendantOf(chassis) then
                if iData.Highlight then iData.Highlight:Destroy() end
                ESP.InternalInstances[part] = nil
            end
        end
        local hull = chassis:FindFirstChild("Hull")
        if hull then for _, obj in ipairs(hull:GetChildren()) do if obj:IsA("Model") then tryRemove(obj) end end end
        local turret = chassis:FindFirstChild("Turret")
        if turret then for _, obj in ipairs(turret:GetChildren()) do if obj:IsA("Model") then tryRemove(obj) end end end
        return
    end

    -- Highlight ESP (hull + turret)
    local hull = chassis:FindFirstChild("Hull")
    if hull then
        for _, obj in ipairs(hull:GetChildren()) do
            if obj:IsA("Model") then
                CreateESP(findHighlightTarget(obj), ESP.HullColor, true)
                break
            end
        end
    end
    local turret = chassis:FindFirstChild("Turret")
    if turret then
        for _, obj in ipairs(turret:GetChildren()) do
            if obj:IsA("Model") then
                CreateESP(findHighlightTarget(obj), ESP.TurretColor, false)
                break
            end
        end
    end

    -- Internal (crew/module) ESP — scan ALL descendants by name, regardless of path
    if ESP.CrewEnabled or ESP.ModulesEnabled then
        ProcessInternalParts(chassis)
    end
end

local function ScanVehicles()
    local vehiclesFolder = workspace:FindFirstChild("Vehicles")
    if not vehiclesFolder then return end
    for _, chassis in ipairs(vehiclesFolder:GetChildren()) do ProcessChassis(chassis) end
end

local function RestoreGraphics()
    Services.Lighting.GlobalShadows = Graphics.OriginalShadows
    for part, mat in pairs(Graphics.OriginalMaterials) do if part and part.Parent then part.Material = mat end end
    table.clear(Graphics.OriginalMaterials)
    for obj, val in pairs(Graphics.OriginalDecals) do
        if obj and obj.Parent then
            if obj:IsA("Decal") or obj:IsA("Texture") then obj.Transparency = val
            else obj.Enabled = val end
        end
    end
    table.clear(Graphics.OriginalDecals)
end

local function DegradeInstance(v, level)
    if level == "Low" then
        return
    end
    if level == "Medium" then
        if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
            if Graphics.OriginalDecals[v] == nil then Graphics.OriginalDecals[v] = v.Enabled end
            v.Enabled = false
        end
        return
    end
    if level == "High" then
        if v:IsA("BasePart") then
            if not Graphics.OriginalMaterials[v] then Graphics.OriginalMaterials[v] = v.Material end
            v.Material = Enum.Material.SmoothPlastic
        elseif v:IsA("Decal") or v:IsA("Texture") then
            if Graphics.OriginalDecals[v] == nil then Graphics.OriginalDecals[v] = v.Transparency end
            v.Transparency = 1
        elseif v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
            if Graphics.OriginalDecals[v] == nil then Graphics.OriginalDecals[v] = v.Enabled end
            v.Enabled = false
        end
    end
end

local function SetGraphicsDegradation(level)
    if Graphics.DegradationLevel == "None" and level ~= "None" then
        Graphics.OriginalShadows = Services.Lighting.GlobalShadows
    end
    
    Graphics.DegradationLevel = level
    if Graphics.AddedConnection then Graphics.AddedConnection:Disconnect(); Graphics.AddedConnection = nil end
    
    RestoreGraphics()
    
    if level == "None" then return end
    
    Services.Lighting.GlobalShadows = false
    for _, v in ipairs(Services.Workspace:GetDescendants()) do DegradeInstance(v, level) end
    
    Graphics.AddedConnection = Services.Workspace.DescendantAdded:Connect(function(v)
        DegradeInstance(v, level)
    end)
end

local function findFlyPart(model)

    for _, part in ipairs(model:GetDescendants()) do

    end
    for _, part in ipairs(candidates) do if part.Name:lower():find("hull") then return part end end

        return candidates[1]
    end
    return nil
end

local function ForceValue(obj, hack)
    if not hack.patched[obj] then hack.patched[obj] = obj.Value end
    if obj:GetAttribute("Orig") ~= nil then
        if not hack.origPatched[obj] then hack.origPatched[obj] = obj:GetAttribute("Orig") end
        obj:SetAttribute("Orig", hack.target)
    end
    obj.Value = hack.target
end

local function WatchWeaponObject(obj)
    for _, hack in pairs(HACKS) do
        if hack.active then
            for _, name in ipairs(hack.names) do
                if obj.Name == name then
                    ForceValue(obj, hack)
                    WeaponConnections[obj] = obj:GetPropertyChangedSignal("Value"):Connect(function()
                        if obj.Value ~= hack.target then ForceValue(obj, hack) end
                    end)
                    return
                end
            end
        end
    end
end

local function ClearWeaponConnections()
    for _, conn in pairs(WeaponConnections) do conn:Disconnect() end
    table.clear(WeaponConnections)
end

local function SetupWeaponChassis(chassis)
    if ChassisDescendantConn then ChassisDescendantConn:Disconnect(); ChassisDescendantConn = nil end
    ClearWeaponConnections()
    for _, obj in ipairs(chassis:GetDescendants()) do
        if obj:IsA("IntValue") or obj:IsA("NumberValue") then WatchWeaponObject(obj) end
    end
    ChassisDescendantConn = chassis.DescendantAdded:Connect(function(obj)
        if obj:IsA("IntValue") or obj:IsA("NumberValue") then WatchWeaponObject(obj) end
    end)
end

local function InitWeapon()
    local chassis = GetOwnChassis(); if chassis then SetupWeaponChassis(chassis) end
end

    sound.SoundId = "rbxassetid://" .. tostring(soundId)
    sound.Volume = 0.5
    sound.Parent = game:GetService("SoundService")
    sound:Play()
    task.spawn(function()
        sound.Ended:Wait()
        sound:Destroy()
    end)
end

local function mkCorner(p, r)
    local c = Instance.new("UICorner"); c.CornerRadius = UDim.new(0, r or 8); c.Parent = p; return c
end

local function theme(inst, bgType, textType, strokeType)
    if bgType then
        inst:SetAttribute("ThemeBG", bgType)
        if bgType == "BG" then inst.BackgroundColor3 = T.BG
        elseif bgType == "Sidebar" then inst.BackgroundColor3 = T.Sidebar
        elseif bgType == "Card" then inst.BackgroundColor3 = T.Card
        elseif bgType == "Hover" then inst.BackgroundColor3 = T.Hover
        elseif bgType == "Stroke" then inst.BackgroundColor3 = T.Stroke
        elseif bgType == "Accent" then inst.BackgroundColor3 = T.Accent
        end
    end
    if textType then
        inst:SetAttribute("ThemeText", textType)
        if textType == "Text" then inst.TextColor3 = T.Text
        elseif textType == "Sub" then inst.TextColor3 = T.Sub
        elseif textType == "Accent" then inst.TextColor3 = T.Accent
        end
    end
    if strokeType then
        inst:SetAttribute("ThemeStroke", strokeType)
        if strokeType == "Stroke" then
            if inst:IsA("UIStroke") then inst.Color = T.Stroke else inst.BackgroundColor3 = T.Stroke end
        elseif strokeType == "Accent" then
            if inst:IsA("UIStroke") then inst.Color = T.Accent else inst.BackgroundColor3 = T.Accent end
        end
    end
    return inst
end

local function mkStroke(p, col, th)
    local s = Instance.new("UIStroke"); s.Color = col or T.Stroke; s.Thickness = th or 1; s.Parent = p
    if col == T.Stroke or col == nil then
        theme(s, nil, nil, "Stroke")
    end
    return s
end

local function ApplyTheme()
    if not Rayfield.ScreenGui then return end
    local win = Rayfield.ScreenGui:FindFirstChild("Window")
    local hasImg = (CustomBgImage ~= "")
    local trans = hasImg and 0.25 or 0
    local sideTrans = hasImg and 0.4 or 0
    local cardTrans = hasImg and 0.45 or 0

    if win then
        local bgImg = win:FindFirstChild("BgImage")
        if bgImg then
            if hasImg then
                bgImg.Image = "rbxassetid://" .. tostring(CustomBgImage)
                bgImg.ImageTransparency = 0.25
            else
                bgImg.Image = ""
                bgImg.ImageTransparency = 1
            end
        end
    end

    for _, inst in ipairs(Rayfield.ScreenGui:GetDescendants()) do
        local bgType = inst:GetAttribute("ThemeBG")
        local textType = inst:GetAttribute("ThemeText")
        local strokeType = inst:GetAttribute("ThemeStroke")
        if bgType then
            if bgType == "BG" then
                inst.BackgroundColor3 = T.BG
                inst.BackgroundTransparency = trans
            elseif bgType == "Sidebar" then
                inst.BackgroundColor3 = T.Sidebar
                inst.BackgroundTransparency = sideTrans
            elseif bgType == "Card" then
                inst.BackgroundColor3 = T.Card
                inst.BackgroundTransparency = cardTrans
            elseif bgType == "Hover" then
                inst.BackgroundColor3 = T.Hover
                inst.BackgroundTransparency = hasImg and 0.5 or 0
            elseif bgType == "Stroke" then
                inst.BackgroundColor3 = T.Stroke
                inst.BackgroundTransparency = 0
            elseif bgType == "Accent" then
                inst.BackgroundColor3 = T.Accent
                if not inst:GetAttribute("IgnoreThemeTrans") then
                    inst.BackgroundTransparency = 0
                end
            end
        end
        if textType then
            if textType == "Text" then inst.TextColor3 = T.Text
            elseif textType == "Sub" then inst.TextColor3 = T.Sub
            elseif textType == "Accent" then inst.TextColor3 = T.Accent
            end
        end
        if strokeType then
            if strokeType == "Stroke" then
                if inst:IsA("UIStroke") then inst.Color = T.Stroke else inst.BackgroundColor3 = T.Stroke end
            elseif strokeType == "Accent" then
                if inst:IsA("UIStroke") then inst.Color = T.Accent else inst.BackgroundColor3 = T.Accent end
            end
        end
    end
end

local function mkDrag(frame, handle)
    local dragging, ds, fp = false, nil, nil
    handle.InputBegan:Connect(function(inp)
        if inp.UserInputType ~= Enum.UserInputType.MouseButton1 and inp.UserInputType ~= Enum.UserInputType.Touch then return end
        dragging = true; ds = inp.Position; fp = frame.Position
        inp.Changed:Connect(function() if inp.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end)
    UserInputSvc.InputChanged:Connect(function(inp)
        if not dragging then return end
        if inp.UserInputType ~= Enum.UserInputType.MouseMovement and inp.UserInputType ~= Enum.UserInputType.Touch then return end
        local d = inp.Position - ds
        frame.Position = UDim2.new(fp.X.Scale, fp.X.Offset + d.X, fp.Y.Scale, fp.Y.Offset + d.Y)
    end)
end

        function tabObj:CreateSlider(data)
            local flag = data.Flag; local sMin = data.Range[1]; local sMax = data.Range[2]
            local sInc = data.Increment or 1; local suf = data.Suffix or ""
            local val = math.clamp(data.CurrentValue or sMin, sMin, sMax)
            local card = Instance.new("Frame"); card.Size = UDim2.new(1,0,0,52)
            card.BackgroundColor3 = T.Card; card.BorderSizePixel = 0; card.Parent = scroll

            valLbl.Font = Enum.Font.Gotham; valLbl.TextSize = 12; valLbl.TextColor3 = T.Accent
            valLbl.TextXAlignment = Enum.TextXAlignment.Right
            local displayVal = (sInc % 1 == 0 and tostring(val) or string.format("%.2f", val))
            valLbl.Text = displayVal..suf; valLbl.Parent = card
            theme(valLbl, nil, "Accent")
            local track = Instance.new("Frame"); track.Name = "SliderTrack"
            track.Size = UDim2.new(1,-24,0,6)
            track.Position = UDim2.new(0,12,0,36); track.BackgroundColor3 = Color3.fromRGB(44,44,58)
            track.BorderSizePixel = 0; track.Parent = card; mkCorner(track, 3)
            local fill = Instance.new("Frame"); fill.Size = UDim2.new((val-sMin)/math.max(sMax-sMin,0.0001),0,1,0)
            fill.BackgroundColor3 = T.Accent; fill.BorderSizePixel = 0; fill.Parent = track; mkCorner(fill, 3)
            theme(fill, "Accent")
            local knob = Instance.new("Frame"); knob.Size = UDim2.new(0,14,0,14)
            knob.Position = UDim2.new(1,-7,0.5,-7); knob.BackgroundColor3 = Color3.fromRGB(234,234,244)
            knob.BorderSizePixel = 0; knob.Parent = fill; mkCorner(knob, 7)
            if flag then Rayfield.Flags[flag] = val end

            local isDragging = false
            local function upd(px)
                local pct = math.clamp((px - track.AbsolutePosition.X) / math.max(track.AbsoluteSize.X, 1), 0, 1)
                local raw = sMin + (sMax - sMin) * pct
                local rounded = math.clamp(math.round((raw - sMin) / sInc) * sInc + sMin, sMin, sMax)
                val = rounded
                if flag then Rayfield.Flags[flag] = val end
                local dv = (sInc % 1 == 0 and tostring(val) or string.format("%.2f", val))
                valLbl.Text = dv..suf
                fill.Size = UDim2.new((val-sMin)/math.max(sMax-sMin,0.0001),0,1,0)
                if data.Callback then data.Callback(val) end
            end

            -- Use a transparent button the size of the whole card for hit detection
            local hit = Instance.new("TextButton"); hit.Size = UDim2.new(1,0,0,22)
            hit.Position = UDim2.new(0,0,0,30); hit.BackgroundTransparency = 1; hit.Text = ""; hit.Parent = card
            hit.InputBegan:Connect(function(inp)
                if inp.UserInputType == Enum.UserInputType.MouseButton1
                or inp.UserInputType == Enum.UserInputType.Touch then
                    isDragging = true
                    upd(inp.Position.X)
                end
            end)
            -- Use global InputChanged with isDragging flag (safe, per-slider)
            local movedConn = UserInputSvc.InputChanged:Connect(function(inp)
                if not isDragging then return end
                if inp.UserInputType == Enum.UserInputType.MouseMovement
                or inp.UserInputType == Enum.UserInputType.Touch then
                    upd(inp.Position.X)
                end
            end)
            local endConn = UserInputSvc.InputEnded:Connect(function(inp)
                if inp.UserInputType == Enum.UserInputType.MouseButton1
                or inp.UserInputType == Enum.UserInputType.Touch then
                    isDragging = false
                end
            end)
            -- Clean up those connections when scroll is destroyed
            scroll.Destroying:Connect(function()
                movedConn:Disconnect()
                endConn:Disconnect()
            end)
        end

safeConnect(Services.UserInput.InputBegan, function(input, gameProcessed)
    if gameProcessed then return end
    if Fly.IsRebinding or (tick() - Fly.LastRebindTime) < 0.5 then return end
    local t = Localization[Localization.CurrentLanguage]
    if input.KeyCode == Keys.Toggle then
        ESP.Enabled = not ESP.Enabled; UpdateAllESPInstances(); SaveConfig()
        playSound(TOGGLE_SOUND_ID)
        Rayfield:Notify({
            Title = "ESP",
            Content = ESP.Enabled and (t.ESPEnabled or "ESP Enabled") or (t.ESPDisabled or "ESP Disabled"),
            Duration = 2
        })
    end
    if input.KeyCode == Keys.Fly then
        playSound(TOGGLE_SOUND_ID)
        if Fly.Active then
            stopFly()
            Rayfield:Notify({ Title = "Fly", Content = t.FlyDisabled, Duration = 2 })
        else
            startFly()
            Rayfield:Notify({ Title = "Fly", Content = t.FlyEnabled, Duration = 2 })
        end
    end
end)

-- Mobile joystick (left half of screen)
if Services.UserInput.TouchEnabled then
    safeConnect(Services.UserInput.TouchStarted, function(touch, gameProcessed)
        if not Fly.Active then return end
        if MobileFly.Touch then return end
        local cam = Services.Workspace.CurrentCamera
        local screenW = cam and cam.ViewportSize.X or 1920
        if touch.Position.X < screenW * 0.5 then
            MobileFly.Touch = touch
            MobileFly.StartPos = Vector2.new(touch.Position.X, touch.Position.Y)
            MobileFly.Delta = Vector2.new(0, 0)
        end
    end)
    safeConnect(Services.UserInput.TouchMoved, function(touch, gameProcessed)
        if not Fly.Active then return end
        if touch ~= MobileFly.Touch then return end
        local dx = touch.Position.X - MobileFly.StartPos.X
        local dy = touch.Position.Y - MobileFly.StartPos.Y
        local raw = Vector2.new(dx, dy)
        if raw.Magnitude > MobileFly.MaxRadius then raw = raw.Unit * MobileFly.MaxRadius end
        MobileFly.Delta = raw / MobileFly.MaxRadius
    end)
    safeConnect(Services.UserInput.TouchEnded, function(touch, gameProcessed)
        if touch == MobileFly.Touch then MobileFly.Touch = nil; MobileFly.Delta = Vector2.new(0,0) end
    end)
end

end)

safeConnect(Services.RunService.Heartbeat, function(dt)
    Timers.VehicleScan += dt
    if Timers.VehicleScan >= Timers.ScanInterval then
        Timers.VehicleScan = 0
        ScanVehicles()
        UpdateAllInternalESP()
    end
    Timers.Cleanup += dt
    if Timers.Cleanup >= Timers.CleanupInterval then Timers.Cleanup = 0; CleanupESP() end
    UpdateDistanceLabels()
end)

safeConnect(Services.RunService.RenderStepped, function(dt)
    Mark.TimeSinceUpdate += dt
    if Mark.TimeSinceUpdate >= Mark.UpdateInterval then Mark.TimeSinceUpdate = 0; UpdateMarkPositions() end
    if Fly.Active and Fly.Root and Fly.Root.Parent then
        local cam = Services.Workspace.CurrentCamera
        local move = Vector3.zero
        if Services.UserInput:IsKeyDown(Enum.KeyCode.W) then move += cam.CFrame.LookVector end
        if Services.UserInput:IsKeyDown(Enum.KeyCode.S) then move -= cam.CFrame.LookVector end
        if Services.UserInput:IsKeyDown(Enum.KeyCode.A) then move -= cam.CFrame.RightVector end
        if Services.UserInput:IsKeyDown(Enum.KeyCode.D) then move += cam.CFrame.RightVector end
        if Services.UserInput:IsKeyDown(Enum.KeyCode.Space) then move += Vector3.new(0,1,0) end
        if Services.UserInput:IsKeyDown(Enum.KeyCode.LeftControl) then move -= Vector3.new(0,1,0) end
        if Services.UserInput.TouchEnabled then
            local jx = MobileFly.Delta.X; local jy = -MobileFly.Delta.Y
            if math.abs(jx) > 0.1 or math.abs(jy) > 0.1 then
                local fl = cam.CFrame.LookVector
                local fr = cam.CFrame.RightVector
                move += fl*jy + fr*jx
            end
        end
        bv.Velocity = move.Magnitude > 0 and move.Unit * Fly.Speed or Vector3.zero
        bg.CFrame = cam.CFrame
    end
end)

-- Weapon watcher (background thread to avoid blocking GUI)
task.spawn(function()
    local vehiclesFolder = Services.Workspace:WaitForChild("Vehicles", 60)
    if not vehiclesFolder then return end
    safeConnect(vehiclesFolder.ChildAdded, function(obj)
        if obj.Name == ("Chassis"..LocalPlayer.Name) then
            for _, hack in pairs(HACKS) do hack.patched = {}; hack.origPatched = {} end
            task.wait(1); SetupWeaponChassis(obj)
        end
    end)
    safeConnect(vehiclesFolder.ChildRemoved, function(obj)
        if not obj:IsA("Model") then return end
        local toRemove = {}
        for target, espData in pairs(ESP.Instances) do
            if target == obj or (espData.Target and espData.Target:IsDescendantOf(obj)) then
                table.insert(toRemove, target)
                if espData.Instance then espData.Instance:Destroy() end
                if espData.DistanceBillboard then espData.DistanceBillboard:Destroy() end
                if espData.MarkBillboard then espData.MarkBillboard:Destroy() end
            end
        end
        for _, tgt in ipairs(toRemove) do ESP.Instances[tgt] = nil end
    end)
end)

local function CleanUpScript()
    for _, conn in ipairs(ScriptConnections) do
        if conn and conn.Connected then conn:Disconnect() end
    end
    table.clear(ScriptConnections)

    stopFly()
    PenView_Stop()
    SetGraphicsDegradation("None")
    SetNoSmoke(false)
    SetNoDeathEffect(false)
    ClearWeaponConnections()
    if ChassisDescendantConn then ChassisDescendantConn:Disconnect(); ChassisDescendantConn = nil end
    ClearAllESP()

    if Rayfield.ScreenGui and Rayfield.ScreenGui.Parent then Rayfield.ScreenGui:Destroy() end
    Rayfield.ScreenGui = nil
    ESPToggle = nil

    local loadingScreen = Services.CoreGui:FindFirstChild("KolyanchikLoading")
    if loadingScreen then loadingScreen:Destroy() end

    local markDisplay = Services.CoreGui:FindFirstChild("MarkDisplay")
    if markDisplay then markDisplay:Destroy() end

    local perfData = Services.CoreGui:FindFirstChild("PerfData")
    if perfData then perfData:Destroy() end
end

local function RunStartup()
    if getgenv then
        if getgenv().CleanUpKolyanchikScript then
            pcall(getgenv().CleanUpKolyanchikScript)
        end
        getgenv().CleanUpKolyanchikScript = CleanUpScript
    end

    LoadConfig()
    ShowLoadingScreen(function()
        ScanVehicles()
        if Other.PenView then task.spawn(PenView_Start) end
        if Graphics.DegradationLevel ~= "None" then task.spawn(function() SetGraphicsDegradation(Graphics.DegradationLevel) end) end
        if AntiLag.NoSmoke then task.spawn(function() SetNoSmoke(true) end) end
        if AntiLag.NoDeathEffect then task.spawn(function() SetNoDeathEffect(true) end) end
        InitWeapon()
        BuildGUI()
        ApplyTheme()
        print("kolyanchik script loaded!")
    end)
end
